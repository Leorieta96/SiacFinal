        this.totalPedidoC = totalPedidoC;

        this.fechaEmision = fechaEmision;

    public Date getFechaRecibido() {
        return fechaRecibido;
    }

    public void setFechaRecibido(Date fechaRecibido) {
        this.fechaRecibido = fechaRecibido;
